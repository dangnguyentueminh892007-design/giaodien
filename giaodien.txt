    <script>
        // Danh sách 8 từ vựng yêu cầu
        const vocabulary = [
            {
                word: "Artificial Intelligence",
                ipa: "/ˌɑː.tɪ.fɪʃ.əl ɪnˈtel.ɪ.dʒəns/",
                type: "Từ phức (Compound Noun)",
                illustration: "🤖",
                example: "Artificial Intelligence helps robots think like humans.",
                translation: "Trí tuệ nhân tạo giúp rô-bốt suy nghĩ giống con người."
            },
            {
                word: "Facial Recognition",
                ipa: "/ˌfeɪ.ʃəl ˌrek.əɡˈnɪʃ.ən/",
                type: "Từ phức (Compound Noun)",
                illustration: "👤📱",
                example: "This phone uses facial recognition to unlock.",
                translation: "Điện thoại này sử dụng nhận diện khuôn mặt để mở khóa."
            },
            {
                word: "Chatbot",
                ipa: "/ˈtʃæt.bɒt/",
                type: "Từ đơn (Noun)",
                illustration: "💬🤖",
                example: "The chatbot answered my question instantly.",
                translation: "Chatbot đã trả lời câu hỏi của tớ ngay lập tức."
            },
            {
                word: "Virtual Reality",
                ipa: "/ˌvɜː.tʃu.əl riˈæl.ə.ti/",
                type: "Từ phức (Compound Noun)",
                illustration: "🥽🎮",
                example: "I play video games using virtual reality.",
                translation: "Tớ chơi trò chơi điện tử bằng công nghệ thực tế ảo."
            },
            {
                word: "Activate",
                ipa: "/ˈæk.tɪ.veɪt/",
                type: "Từ đơn (Verb)",
                illustration: "🔘⚡",
                example: "You can say 'Hey Google' to activate the phone.",
                translation: "Cậu có thể nói 'Hey Google' để kích hoạt điện thoại."
            },
            {
                word: "Smart app",
                ipa: "/smɑːt æp/",
                type: "Từ phức (Noun Phrase)",
                illustration: "📱✨",
                example: "This smart app tracks how many steps you walk.",
                translation: "Ứng dụng thông minh này theo dõi số bước chân cậu đi."
            },
            {
                word: "Platform",
                ipa: "/ˈplæt.fɔːm/",
                type: "Từ đơn (Noun)",
                illustration: "💻🌐",
                example: "YouTube is a popular platform for watching videos.",
                translation: "YouTube là một nền tảng phổ biến để xem video."
            },
            {
                word: "Technician",
                ipa: "/tekˈnɪʃ.ən/",
                type: "Từ đơn (Noun)",
                illustration: "👨‍🔧🛠️",
                example: "The technician fixed our broken computer.",
                translation: "Chú kỹ thuật viên đã sửa chiếc máy tính bị hỏng của chúng tớ."
            }
        ];

        let currentIndex = 0;

        // Lấy các phần tử DOM
        const wordType = document.getElementById('wordType');
        const wordTarget = document.getElementById('wordTarget');
        const wordIpa = document.getElementById('wordIpa');
        const wordIllustration = document.getElementById('wordIllustration');
        const wordExample = document.getElementById('wordExample');
        const wordTranslation = document.getElementById('wordTranslation');
        const feedbackMsg = document.getElementById('feedbackMsg');
        const textHeard = document.getElementById('textHeard');
        
        const micBtn = document.getElementById('micBtn');
        const prevBtn = document.getElementById('prevBtn');
        const nextBtn = document.getElementById('nextBtn');
        
        const animal1 = document.getElementById('animal1');
        const animal2 = document.getElementById('animal2');
        const animal3 = document.getElementById('animal3');

        // Hàm cập nhật giao diện khi đổi từ
        function updateCard() {
            const currentItem = vocabulary[currentIndex];
            wordType.innerText = currentItem.type;
            wordTarget.innerText = currentItem.word;
            wordIpa.innerText = currentItem.ipa;
            wordIllustration.innerText = currentItem.illustration;
            wordExample.innerText = `"${currentItem.example}"`;
            wordTranslation.innerText = `(${currentItem.translation})`;
            
            // Trả trạng thái rừng về chờ đợi
            resetForestState();
        }

        function resetForestState() {
            document.body.className = "waiting";
            feedbackMsg.innerText = "Hãy nhấn mic và đọc to từ trên nhé!";
            feedbackMsg.style.color = "#1e293b";
            textHeard.innerText = "";
            animal1.innerText = "🦊";
            animal2.innerText = "🦁";
            animal3.innerText = "🐰";
        }

        // Cấu hình Web Speech API nhận diện giọng nói
        const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
        
        if (!SpeechRecognition) {
            feedbackMsg.innerText = "⚠️ Trình duyệt của bạn không hỗ trợ nhận diện giọng nói. Hãy dùng Google Chrome nhé!";
            micBtn.style.display = "none";
        } else {
            const recognition = new SpeechRecognition();
            recognition.lang = 'en-US'; // Nhận diện tiếng Anh
            recognition.interimResults = false;
            recognition.maxAlternatives = 1;

            micBtn.addEventListener('click', () => {
                recognition.start();
                micBtn.classList.add('recording');
                micBtn.innerText = "🎧 Đang nghe nè...";
                resetForestState();
            });

            recognition.onresult = (event) => {
                micBtn.classList.remove('recording');
                micBtn.innerText = "🎙️ Nhấn để đọc";
                
                // Lấy kết quả từ bé nói (chuyển về chữ thường, xóa dấu chấm để dễ so sánh)
                let speechResult = event.results[0][0].transcript.toLowerCase().trim().replace(/[.,\/#!$%\^&\*;:{}=\-_`~()]/g,"");
                let targetWord = vocabulary[currentIndex].word.toLowerCase().trim();

                textHeard.innerText = `Bạn vừa nói: "${event.results[0][0].transcript}"`;

                // Kiểm tra xem bé nói có đúng với từ mục tiêu không
                if (speechResult === targetWord || speechResult.includes(targetWord)) {
                    // Khi nói ĐÚNG
                    document.body.className = "success";
                    feedbackMsg.innerText = "🎉 Tuyệt vời! Bạn phát âm chuẩn lắm! ✨";
                    feedbackMsg.style.color = "#047857";
                    
                    // Động vật mỉm cười, vui vẻ
                    animal1.innerText = "🤩";
                    animal2.innerText = "😸";
                    animal3.innerText = "🥳";
                } else {
                    // Khi nói SAI
                    document.body.className = "wrong";
                    feedbackMsg.innerText = "😢 Gần đúng rồi! Thử lại một lần nữa xem sao nhé!";
                    feedbackMsg.style.color = "#b91c1c";
                    
                    // Động vật buồn bã
                    animal1.innerText = "😭";
                    animal2.innerText = "😿";
                    animal3.innerText = "🥺";
                }
            };

            recognition.onspeechend = () => {
                recognition.stop();
                micBtn.classList.remove('recording');
                micBtn.innerText = "🎙️ Nhấn để đọc";
            };

            recognition.onerror = (event) => {
                micBtn.classList.remove('recording');
                micBtn.innerText = "🎙️ Nhấn để đọc";
                feedbackMsg.innerText = "⚠️ Tớ chưa nghe rõ, bạn bấm mic nói lại nha!";
            };
        }

        // Sự kiện nút chuyển từ
        nextBtn.addEventListener('click', () => {
            currentIndex = (currentIndex + 1) % vocabulary.length;
            updateCard();
        });

        prevBtn.addEventListener('click', () => {
            currentIndex = (currentIndex - 1 + vocabulary.length) % vocabulary.length;
            updateCard();
        });

        // Khởi tạo từ đầu tiên khi load trang
        updateCard();
    </script>
